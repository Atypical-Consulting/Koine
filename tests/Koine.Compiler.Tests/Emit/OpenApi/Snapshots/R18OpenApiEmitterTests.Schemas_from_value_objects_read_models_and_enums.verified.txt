info:
  title: "Catalog API"
  version: 1.0.0
paths: {}
components:
  schemas:
    Currency:
      type: string
      description: "A supported settlement currency."
      enum:
        - EUR
        - USD
    ItemRow:
      type: object
      properties:
        sku:
          type: string
        price:
          $ref: "#/components/schemas/Money"
      required:
        - sku
        - price
    Money:
      type: object
      properties:
        amount:
          type: number
        currency:
          $ref: "#/components/schemas/Currency"
      required:
        - amount
        - currency
    Product:
      type: object
      properties:
        sku:
          type: string
        name:
          type: string
        price:
          $ref: "#/components/schemas/Money"
        tags:
          type: array
          items:
            type: string
        discount:
          type:
            - number
            - "null"
      required:
        - sku
        - name
        - price
        - tags
